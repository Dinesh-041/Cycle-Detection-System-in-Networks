import java.util.*;

public class Main {
   
    enum NodeState {
        UNVISITED,
        VISITING,
        VISITED
    }

    private final Map<String, List<String>> adjacencyList;

    public Main() {
        this.adjacencyList = new HashMap<>();
    }

    public void addLink(String sourceRouter, String destinationRouter) {
        this.adjacencyList.putIfAbsent(sourceRouter, new ArrayList<>());
        this.adjacencyList.putIfAbsent(destinationRouter, new ArrayList<>());
        this.adjacencyList.get(sourceRouter).add(destinationRouter);
    }

    public boolean hasRoutingLoop() {
        Map<String, NodeState> stateMap = new HashMap<>();
        for (String router : adjacencyList.keySet()) {
            stateMap.put(router, NodeState.UNVISITED);
        }

        for (String router : adjacencyList.keySet()) {
            if (stateMap.get(router) == NodeState.UNVISITED) {
                if (dfsCheckCycle(router, stateMap)) {
                    return true;
                }
            }
        }
        return false;
    }

    private boolean dfsCheckCycle(String currentRouter, Map<String, NodeState> stateMap) {
        stateMap.put(currentRouter, NodeState.VISITING);

        List<String> neighbors = adjacencyList.getOrDefault(currentRouter, new ArrayList<>());
        for (String neighbor : neighbors) {
            NodeState neighborState = stateMap.getOrDefault(neighbor, NodeState.UNVISITED);

            if (neighborState == NodeState.VISITING) {
                System.out.println("\n Loop Detected! Backlink found from " + currentRouter + " -> " + neighbor);
                return true;
            }

            if (neighborState == NodeState.UNVISITED) {
                if (dfsCheckCycle(neighbor, stateMap)) {
                    return true;
                }
            }
        }

        stateMap.put(currentRouter, NodeState.VISITED);
        return false;
    }

    public static void main(String[] args) {
        Main network = new Main();
      
        Scanner scanner = new Scanner(System.in);

        System.out.println("=== Network Loop Detector ===");
        
       
        int totalLinks = 0;
        while (true) {
            System.out.print("Enter the total number of network links to add: ");
            if (scanner.hasNextInt()) {
                totalLinks = scanner.nextInt();
                if (totalLinks >= 0) break;
                System.out.println("Please enter a positive number.");
            } else {
                System.out.println("Invalid input. Please enter an integer.");
                scanner.next(); // Clear invalid token
            }
        }

       
        for (int i = 1; i <= totalLinks; i++) {
            System.out.println("\n--- Link " + i + " ---");
            System.out.print("Enter Source Router Name (e.g., Router_A): ");
            String source = scanner.next();
            
            System.out.print("Enter Destination Router Name (e.g., Router_B): ");
            String destination = scanner.next();
            
            network.addLink(source, destination);
        }

        scanner.close();

       
        System.out.println("\nAnalyzing network topology for routing loops...");
       
        if (network.hasRoutingLoop()) {
            System.out.println(" Network Alert: Malformed topology! Cycle detected.");
        } else {
            System.out.println(" Network Stable: No cycles detected.");
        }
    }
}
